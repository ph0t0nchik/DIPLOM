git status - что у нас в локальном репозитории, изменения выделены красным

git add . - с точкой, то добавить все файлы в локальный репозиторий

git add filename.ext

git status

git commit -m "add filename.ext" - подготовили изменения для глобального репозитори

git push -u origin main - отправили файл в глобальный репозиторий
